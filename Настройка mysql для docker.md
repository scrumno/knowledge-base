Обязательно создаем папку /mysql на локале, она нужна для того, что бы мы загрузили туда всё необходимое для базы данных

```bash
mkdir mysql
```

Минимальная настройка сервиса mysql ^610bd3

```yml
# Определяем сервис с именем mysql
# Настраиваем контейнер для базы данных MySQL
  mysql:
  
	# Указываем образ для сервиса mysql
	# mysql:8.0 => официальный образ из Docker Hub
    image: mysql:8.0
    
    # Пробрасываем порты для MySQL
    # Первый аргумент => порт хоста (наш локальный)
    # Второй аргумент => порт контейнера (в данном случае, стандарный порт MySQL)
    # Так мы можем подключаться к MySQL с хоста (с нашего компьютера или сервера) через mysql -h localhost -P 3306
    ports:
      - 3306:3306
        
    # Задаем переменные окружения для MySQL
    # В данном случае создаем БД, пользователя, устанавливаем ему пароль и устанавливаем пароль для root-пользователя
    environment:
      MYSQL_DATABASE: laravel
      MYSQL_USER: laravel
      MYSQL_PASSWORD: secret
      MYSQL_ROOT_PASSWORD: secret
    
    # Настраиваем MySQL с указанными учетными данными и базой данных при старте контейнера
    # Монтируем том для хранения данных MySQL
    # То-есть здесь мы связываем локальную директорию ./mysql с /var/lib/mysql в контейнере (где MySQL хранит данные базы)
    # Так данные базы сохраняются на хосте, обеспечивая персистентность между перезапусками контейнера
    volumes:
      - ./mysql:/var/lib/mysql
    
    # Делаем проверку, "здоров ли контейнер", это нужно для того, что бы, например, artisan делал миграции корректно 
    healthcheck:
    
	  # Тест указывает команду для проверки состояния
	  # Кратко: проверяет, отвечает ли MySQL-сервер. Если сервер доступен, возвращает heathly
	  # *-h localhost => означает, что проверка делается на локальном хосте внутри контейнера
	  # Проверка выполняется каждые 10 секунд, если проверка неуспешна, докер повторяет попытку 5 раз, прежде чем пометить сервис как unhealthy
      test: ['CMD', 'mysqladmin', 'ping', '-h', 'localhost']
      interval: 10s
      retries: 5
```
